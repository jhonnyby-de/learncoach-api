Quick Start

# Generate a smart learning plan
curl -X POST https://api.learncoach.com/api/planner/generate \\
  -H "Content-Type: application/json" \\
  -d '{
    "grade": 9,
    "subjects": ["Mathematik", "Deutsch", "Englisch"],
    "targetNote": 2.0,
    "days": 30,
    "hoursPerDay": 2,
    "currentLevels": {
      "math_g9_001": 2.5,
      "math_g9_002": 3.0
    }
  }'
